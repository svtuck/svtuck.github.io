Python vs Java Why Python vs. Java? Both awesome.

Python is:
 - easy (relatively) to learn
 - easy to write code quickly: can say a lot with a little
 - designed for writing relatively small programs
 - indenting matters: makes code easy to read
 - no declaration of variables. It kind of "figures things out" while
   program is running; easier to code to a point, but if you misspell
   a variable, never find out until program is running, and only if
   you walk down that path
 - slow as molasses unless you use crazy tricks
 - doesn't encapsulate very well for writing large software systems


Java is: ... the opposite to most of that

package day1;

import java.io.File;
import java.io.IOException;
import java.nio.file.Files;
import java.util.Scanner;

public class Welcome {

    public static void main(String [] args) throws IOException {
    	// Hello World
        System.out.println("Hello world");

        // Read Arguments
        System.out.println("Got " + args.length + " arguments");
        for(int i = 0; i < args.length ; i ++) {
            System.out.println("Argument " + i + " is " + args[i]);
        }

        // Read from standardin
        Scanner scanner = new Scanner(System.in);
        System.out.println("Enter a file: ");

        String input = scanner.nextLine();

        // Read file 1
        for(String line : Files.readAllLines(new File(input).toPath())) {
            System.out.println(line);
        }

        // Read file 2
        Scanner fileScanner = new Scanner(new File(input));
        while( fileScanner.hasNextLine()) {
            System.out.println(fileScanner.nextLine());
        }


        // Random number
        int secret = (int)((Math.random()*100)+1);
        Scanner keyboard = new Scanner(System.in);

        System.out.print("What is your guess? ");
        int guess = keyboard.nextInt();
        boolean success = testTheGuess(guess, secret);

        while (!success)
        {
            System.out.print("What is your guess? ");
            guess = keyboard.nextInt();
            success = testTheGuess(guess, secret);
        }
    }



    public static boolean testTheGuess(int guess, int secret)
    {
        if (guess < secret)
        {
            System.out.println("Too low, try again.");
            return false;
        }
        else if (guess > secret)
        {
            System.out.println("Too high, try again");
            return false;
        }
        else
        {
            System.out.println("You got it!");
            return true;
        }
    }

}
